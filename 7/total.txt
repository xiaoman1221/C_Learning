#include <stdio.h>
#include <stdlib.h>

void main()
{
    printf("欢迎来到本系统\n请按照以下提示完成对系统的操作\n");
    printf("[1]增\n");
    printf("[2]删\n");
    printf("[3]改\n");
    printf("[4]查\n");
    printf("[0]退出本程序\n");
    printf("请输入：");
    int num;
    scanf("%d", &num);
    getchar();
    switch (num)
    {
        case 0:
        printf("感谢您使用本系统，再见");
        exit(0);
        case 1:
        printf("增加了一条数据");
        break;
        case 2:
        printf("删除了一条数据");
        break;
        case 3:
        printf("更新了一条数据");
        break;
        case 4:
        printf("这是一条数据");
        break;
    default:
        printf("未知的输入！！，程序退出");
        exit(1);
    }
}
