#include <stdio.h>
int main()
{
    int numbers[5] = {10,20,30,40,50};
    for(int i=0;i<5;i++)
    {
        printf("%d,",numbers[i]);
    }
    printf("\n");
    char a[] = {'a','e','i','o','u'};
    for(int i=0;i<5;i++)
    {
        printf("%c",a[i]);
    }
    printf("\n");
    float b[] = {11.4,5.14,19.10};
    for(int i=0;i<10;i++)
    {
        printf("%.1f",b[i]);
    }
    printf("\n");
    return 0;
}

#include <stdio.h>

{
    int arr[5]={1,2,3,4,5};
    printf("第三个元素：%d\n",arr[0]);
    printf("第三个元素：%d\n",arr[2]);

    arr[0] = 20;
    printf("遍历所有数组");
    for (int i = 0; i < 5; i++)
    {
        printf("%d ",arr[i]);
    }
    printf("\n");
    return 0;
}

{
