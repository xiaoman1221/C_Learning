#include <stdio.h>

void main()
{
    char a;
    double x,y;
    printf("输入+,-,*,/ ：");
    scanf("%c",&a);
    getchar();
    printf("输入两个数字：");
    scanf("%lf%lf",&x,&y);
    getchar();
    if(a=='+')
    {
        printf("%.2lf+%.2lf=%.2lf\n",x,y,x+y);
    }else if(a=='-')
    {
        printf("%.2lf-%.2lf=%.2lf\n",x,y,x-y);
    }else if(a=='*')
    {
        printf("%.2lf*%.2lf=%.2lf\n",x,y,x*y);
    }else if(a=='/')
    {
        if(y==0)
        {
            printf("除数不为0");
        }else
        {
            printf("%.2lf/%.2lf=%.2lf\n",x,y,x/y);
        }
    }else
    {
        printf("未知运算符");
    }
}
