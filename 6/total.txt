#include <stdio.h>

void main()
{
    int sec;
    printf("输入安全等级: ");
    scanf("%d", &sec);
    getchar();
    switch(sec)
    {
        case 1:
            printf("安全等级为：正常");
            break;
        case 2:
            printf("安全等级：警报！需要注意");
            break;
        case 3:
            printf("安全等级：危险！请立即终止");
            break;
        case 4:
            printf("安全等级 ：立即停产");
            break;
        default:
            printf("输入有问题：请重新运行此程序！程序已退出");
            break;
    }
}
